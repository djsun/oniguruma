      def match_all string
         matches = []
         positions = []
         position = 0
         tmp_string = string
         while tmp_string != ""
            if m = match( tmp_string )
               matches << m
               positions << position
               tmp_string = m.post_match
               position += m.end
               #if m.end == m.begin
               #   tmp_string = tmp_string[1..-1]
               #   position += 1
               #end
            else
               break
            end
         end
         if matches.size > 0
            MultiMatchData.new( string, matches, positions )
         else
            nil
         end
      end
